Public Sub report_to_gen()

    strprev = Format(commonprevbusdate, "yyyy-mm-dd")
    repyear2int = Val(repyear)
    date1 = repmonth & " " & repday & ", " & repyear
    number = Val(repdatenum)
    
    
    repdatenum2 = Str(number)
    
    repmonth2 = Switch(number = 1, "January", number = 2, "February", number _
                    = 3, "March", number = 4, "April", number = 5, "May", number _
                    = 6, "June", number = 7, "July", number = 8, "August", number _
                    = 9, "September", number = 10, "October", number = 11, "November", number = 12, "December")
    
    'set month table
    findterm
    
    'set database here
    ADO_Connect
    Clear "month"
    conn.Execute ("insert into month(term) values('" & Str(month) & "')")
    Clear "branch_to"
    conn.Execute ("insert into branch_to(branchno) values('" & txtBranch.Text & "')")
    Clear "prevbusdate"
    conn.Execute ("insert into prevbusdate(date) values('" & strprev & "')")
    ADO_Disconnect


End Sub

Public Sub Clear(table As String)

On Error GoTo Error_found
DoEvents

    conn.CommandTimeout = 0
    conn.Execute "Delete from " & table

    
    Exit Sub
    
Error_found:
If Err.number <> 0 Then MsgBox Err.Description, vbExclamation, "ERROR"

End Sub
